# 通过实例学习reduce函数

reduce() 方法对数组中的每个元素执行一个由您提供的reducer函数(升序执行)，将其结果汇总为单个返回值。

arr.reduce(callback(accumulator, currentValue[, index[, array]])[, initialValue])

reducer 函数接收4个参数:

  • Accumulator (acc) (累计器)
  • Current Value (cur) (当前值)
  • Current Index (idx) (当前索引)
  • Source Array (src) (源数组)

# 实例1

计算数组[0, 1, 2, 3]里所有值的和


const arr = [0, 1, 2, 3];
const total = arr.reduce((accumulatorValue, currentValue) => accumulatorValue + currentValue, 0); // total = 6

# 实例4

var names = ['Alice', 'Bob', 'Tiff', 'Bruce', 'Alice']; 计算数组中每个元素出现的次数；


const arr = ['Alice', 'Bob', 'Tiff', 'Bruce', 'Alice'];
arr.reduce((accumulatorValue, currentItem) => {
    if(accumulatorValue[currentItem]){
        accumulatorValue[currentItem]++
    }else{
        accumulatorValue[currentItem] = 1;
    }

    return accumulatorValue;
}, {}); 
// {Alice: 2, Bob: 1, Tiff: 1, Bruce: 1}

# 实例5

var people = [ { name: 'Alice', age: 21 }, { name: 'Max', age: 20 }, { name: 'Jane', age: 20 } ];

按属性对object分类

function groupBy(array, property){
    return array.reduce((accumulatorObject, currentObject) => {
        if(!accumulatorObject[currentObject[property]]){
            accumulatorObject[currentObject[property]] = [];
        }
        
        accumulatorObject[currentObject[property]].push(currentObject);

        return accumulatorObject;
    }, {})
}

const people = [
  { name: 'Alice', age: 21 },
  { name: 'Max', age: 20 },
  { name: 'Jane', age: 20 }
];

groupBy(people, "name");
/**
 * {
 *     Alice: [{ name: 'Alice', age: 21 }],
 *     Jane: [{ name: 'Jane', age: 20 }]
 *     Max: [{ name: 'Max', age: 20 }],
 * }
*/

groupBy(people, "age");
/**
 * {
 *     20: [{ name: 'Max', age: 20 }, { name: 'Jane', age: 20 }]
 *     21: [{ name: 'Alice', age: 21 }],
 * }
*/

# 实例6

var friends = [{ name: 'Anna', books: ['Bible', 'Harry Potter'], age: 21 }, { name: 'Bob', books: ['War and peace', 'Romeo and Juliet'], age: 26 }, { name: 'Alice', books: ['The Lord of the Rings', 'The Shining'], age: 18 }];

EXCEPT: allbooks = [ 'Alphabet', 'Bible', 'Harry Potter', 'War and peace', 'Romeo and Juliet', 'The Lord of the Rings', 'The Shining' ]

使用扩展运算符和initialValue绑定包含在对象数组中的数组


const friends = [{
  name: 'Anna',
  books: ['Bible', 'Harry Potter'],
  age: 21
}, {
  name: 'Bob',
  books: ['War and peace', 'Romeo and Juliet'],
  age: 26
}, {
  name: 'Alice',
  books: ['The Lord of the Rings', 'The Shining'],
  age: 18
}];

friends.reduce((accumulatorArray, currentObject) =>{
    return [...accumulatorArray, ...currentObject.books]
}, ['Alphabet']);
//  ['Alphabet', 'Bible', 'Harry Potter', 'War and peace', 'Romeo and Juliet', 'The Lord of the Rings', 'The Shining']

# 实例7

数组 ['a', 'b', 'a', 'b', 'c', 'e', 'e', 'c', 'd', 'd', 'd', 'd'] 去重


const array = ['a', 'b', 'a', 'b', 'c', 'e', 'e', 'c', 'd', 'd', 'd', 'd'];
array.reduce((accumulatorArray, currentItem) => {
    if(accumulatorArray.indexOf(currentItem) === -1){
        accumulatorArray.push(currentItem)
    }

    return accumulatorArray
}, []);
//  ['a', 'b', 'c', 'e', 'd']
