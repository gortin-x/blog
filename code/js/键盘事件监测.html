# 键盘事件监测

js监测键盘事件在安卓和ios环境中的表现并不一致，安卓键盘弹起时会改变当前窗口的高度，因此可以通过resize 监测窗口变化来实现。而在ios环境中，键盘事件并不会影响当前窗口，但是ios本身提供了focusin/focusout这对方法来通知键盘事件。

export default function keyboardEvent(watcher?: Function) {
    if (!document) {
        throw new Error("Error in getKeyboardStatus, document not exit!");
    }

    let status: boolean = false,
        _watcher: Function | undefined = watcher;
    const originHeight: number = document.documentElement.clientHeight || document.body.clientHeight;

    if (this.isFunction(_watcher)) {
        _bind.apply(this)
    }

    function _bind(): void {
        // if ios
        if (/*FIXME isIOS */) {
            document.body.addEventListener('focusin', () => {
                // 键盘弹起
                status = true;
                _watcher && _watcher(status)
            });
            document.body.addEventListener('focusout', () => {
                // 键盘收起
                status = false;
                _watcher && _watcher(status)
            });
        }
        // if android
        else {
            if (!window) {
                throw new Error("Error in getKeyboardStatus, window not exit!");
            }

            window.addEventListener('resize', _resize);
        }

        /**
         * 定制化代码
         * Hybird 开发可能会用到
         * 根据Native提供的协议定制化处理
         * */
        // if (/*定制化环境*/) {
        //     (window as any)["keyboardAction"] = (result: string): void => {
        //         if (result == 'keyboardUp') {
        //             // 键盘弹起
        //             status = true;
        //         } else if (result === 'keyboardDown') {
        //             // 键盘收起
        //             status = false;
        //         }

        //         _watcher && _watcher(status)
        //     }
        }

    }

    function _resize(): void {
        const resizeHeight: number = document.documentElement.clientHeight || document.body.clientHeight;
        if (resizeHeight < originHeight) {
            // 键盘弹起
            status = true;
        } else {
            // 键盘收起
            status = false;
        }

        _watcher && _watcher(status)
    }

    function _get(): boolean {
        return status;
    }

    return {
        get: _get
    }
};

    // status : true 键盘打开 ｜ false 键盘收起

})


// 2、主动获取
keyboardEvent().get() // true 键盘打开 ｜ false 键盘收起
